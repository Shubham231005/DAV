x1<- c(4,3,1.6,1.2,3.4,4.8)
x2<-c(1.5,2.2,1.0,2.0,0.8,1.6)
y<-c(160,112,69,90,123,186)
data<-data.frame(x1,x2,y)
model<-lm(y~x1+x2,data = data)
a<-coef(model)[1]
b1<-coef(model)[2]
b2<-coef(model)[3]
cat("intercept(a)=",a,"\n")
cat("coefficient b1=",b1,"\n")
cat("coefficient b2=",b2,"\n")
cat("regression equation:\n")
cat("y=",round(b1,3),"*x1+",round(b2,3),"*x2\n\n")
x1_new<-3.7
x2_new<-2.6
y_pred<-a+b1*x1_new+b2*x2_new
cat("predicted price = ",round(y_pred,3),"\n")
